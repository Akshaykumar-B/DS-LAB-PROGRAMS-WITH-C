#include <stdio.h>

int main() {
    int n, i, j;
    
    printf("Enter size of array: ");
    scanf("%d", &n);
    
    int arr[n];
    printf("Enter %d elements: ", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    printf("Elements repeated twice are: ");
    for (i = 0; i < n; i++) {
        int count = 0;
        for (j = 0; j < n; j++) {
            if (arr[i] == arr[j]) {
                count++;
            }
        }
        // print only if repeated exactly 2 times and not printed before
        if (count == 2) {
            int alreadyPrinted = 0;
            for (j = 0; j < i; j++) {
                if (arr[i] == arr[j]) {
                    alreadyPrinted = 1;
                    break;
                }
            }
            if (!alreadyPrinted) {
                printf("%d ", arr[i]);
            }
        }
    }
    
    return 0;
}
